2.2 Data preprocessing

  • Convert data to returns
  • Generate some descriptive statistics
  • Some plots
# using close prices
bhp2 = BHP$BHP.AX.Close
asx2 = ASX$AXJO.Close
# covert to returns

bhp_ret = dailyReturn(bhp2, type = "log")
asx_ret = dailyReturn(asx2, type = "log")
# check number of days they must be the same

setdiff(index(bhp_ret), index(asx_ret))  #should return 0 or else we will need to find common dates
numeric(0)
# create a data frame with values and dates

data_lm1 = data.frame(index(bhp_ret), bhp_ret$daily.returns, asx_ret$daily.returns)
# change column names
colnames(data_lm1) = c("Date", "bhp", "asx")
